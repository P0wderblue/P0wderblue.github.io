启动namesrv

nohup ./mqnamesrv -n 47.113.204.36:9876 > logs/mqnamesrv.out 2>1 &

查看日志
tail -f logs/mqnamesrv.out
启动broker
nohup sh ./mqbroker -n 47.113.204.36:9876 autoCreateTopicEnable=true > logs/mqbroker.out 2>1 &
查看日志
tail -f logs/mqbroker.out

./mqshutdown broker
./mqshutdown namesrv

sh startup.sh -m standalone
